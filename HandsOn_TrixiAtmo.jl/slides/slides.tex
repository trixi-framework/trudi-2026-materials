\begin{frame}{Physical background}
% [Marco]

Main features of Atmospheric flows
\begin{itemize}
\item Rotation and geopotential, $f = 2 \Omega \times \vec{V}$.
\item Gravity and hydrostatic balance $\nabla p + \varrho \nabla \phi = 0$.
% -> figures/hydrostatic.pdf
\item Curvature (e.g. curvilinear or covariant formulations)
\item Disparity between vertical and horizontal scales
\begin{columns}[T] % T = top alignment
    \column{0.7\textwidth} % left column: image
    \begin{figure}
        \centering
        \includegraphics[scale=0.12]{figures/mesh_cubedsphere.png}
    \end{figure}

    \column{0.3\textwidth}
    \vspace{1cm} % right column: equation
\begin{equation*}
\begin{aligned}
& \Delta x \approx 1700 \text{km}\\
& \Delta z \approx 7500 \text{km}
\end{aligned}
\end{equation*}
\end{columns}
\item Sound waves and low Mach numbers
 \begin{equation}
 \text{CFL} \approx \frac{\max (\lambda)  \Delta t}{\Delta x}, \quad \text{where} \max (\lambda) \approx c \ll |\vec{V}|.
 \end{equation}
\item Interfaces
\end{itemize}

\end{frame}

\begin{frame}{Session 2: Vector invariant form}
% [Marco]
Implement the compressble Euler equations in vector invariant form
\begin{equation*}
  \begin{aligned}
&\frac{\partial \rho}{\partial t}
+
\nabla \cdot (\rho \mathbf{V})
=0\\
&\frac{\partial \mathbf{V}}{\partial t}
+
(\nabla \times \mathbf{V}) \times \mathbf{u}
+
\nabla\left(\frac{|\mathbf{V}|^2}{2}\right)
+
\theta \nabla \Pi
+
 \nabla \phi
=0\\
&\frac{\partial (\rho\theta)}{\partial t}
+
\nabla \cdot (\rho\theta \mathbf{u})
=0\\
  \end{aligned}
\end{equation*}
where $p = p_0 \left ( \frac{\varrho \theta R}{p_0} \right )^{\gamma}$ and $\Pi = \left (\frac{\varrho \theta R}{p_0} \right )^{R/{c_v}}$.

\begin{equation*}
  \begin{aligned}
&\frac{\partial \rho}{\partial t}
+ \frac{\partial (\rho u)}{\partial x}
+ \frac{\partial (\rho v)}{\partial y}
=0\\
&\frac{\partial u}{\partial t}
+ v\left(\frac{\partial u}{\partial y}-\frac{\partial v}{\partial x}\right)
+ \frac{\partial}{\partial x}\left(\frac{u^2+v^2}{2}\right)
+ \theta \frac{\partial \Pi}{\partial x}
+ \frac{\partial \phi}{\partial x}
=0\\
&\frac{\partial v}{\partial t}
+ u\left(\frac{\partial v}{\partial x}-\frac{\partial u}{\partial y}\right)
+ \frac{\partial}{\partial y}\left(\frac{u^2+v^2}{2}\right)
+ \theta \frac{\partial \Pi}{\partial y}
+ \frac{\partial \phi}{\partial y}
=0\\
&\frac{\partial (\rho\theta)}{\partial t}
+ \frac{\partial (\rho\theta u)}{\partial x}
+ \frac{\partial (\rho\theta v)}{\partial y}
=0
  \end{aligned}
\end{equation*}

A basic code implementing the strictly necessary components is already provided. Be careful that different approaches may heavily affect the \textbf{performance}.
\end{frame}

Nonconservative numerical flux
\begin{lstlisting}[language=Julia]
@inline function flux_volume_nonconservative(u_ll, u_rr, 
                            orientation, equations)
      
      _, H_ll, g_ll = cons2whateverIneed(u_ll, equations)
      _, H_rr, g_rr = cons2whateverIneed(u_rr, equations)                      
  
      H_mean = 0.5f0 * (H_ll + H_rr)
      jump_g = g_rr - g_ll
  
#(H^num_{i+1/2} (g_i+1 - g_i) + H^num_{i-1/2} (g_i - g_i-1))/(2 Delta x)
      f1_noncons = H_mean * jump_g 
  
# H_i (g_i+1 - g_i-1)/(2 Delta x)
      f1_noncons = H_ll * g_rr       

    return f1_noncons
end

\end{lstlisting}

\end{frame}

\begin{lstlisting}[language=Julia, basicstyle=\scriptsize\ttfamily]
@inline function flux_volume_combined(u_ll, u_rr, 
                            orientation, equations)
      
      f_ll, H_ll, g_ll = cons2whateverIneed(u_ll, equations)
      f_rr, H_rr, g_rr = cons2whateverIneed(u_rr, equations)                      
  
  f1 = (f_ll + f_rr)*0.5f0

      H_mean = 0.5f0 * (H_ll + H_rr)
      jump_g = g_rr - g_ll
  
#(H^num_{i+1/2} (g_i+1 - g_i) + H^num_{i-1/2} (g_i - g_i-1))/(2 Delta x)
      f1_noncons = H_mean * jump_g 
  
# H_i (g_i+1 - g_i-1)/(2 Delta x)
      f1_noncons = H_ll * g_rr       

    f1_noncons_left = flux_nonconservative(u_ll, u_rr)
    f1_noncons_right = flux_nonconservative(u_rr, u_ll)

    return SVector(f1 + 0.5 * f1_noncons_left), 
           SVector(f1 + 0.5 * f1_noncons_right)
end
\end{lstlisting}
